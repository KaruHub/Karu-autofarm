-- Load the latest version of the Orion library
local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/jensonhirst/Orion/main/source')))()

-- Base64 encoded key "Karu!100"
local encodedKey = "S2FydSUhMTAw" -- This is "Karu!100" encoded in Base64

-- Function to decode the Base64 encoded key
local function decodeBase64(input)
    local b='ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    input = string.gsub(input, '[^'..b..'=]', '')
    return (input:gsub('.', function(x)
        if x == '=' then return '' end
        local r, bits = '', string.find(b, x) - 1
        for i = 6, 1, -1 do
            r = r..(bits % 2 ^ i - bits % 2 ^ (i - 1) // 2 ^ (i - 1))
        end
        return r
    end)):match('^(%d+)$')
end

-- Decode the encoded key
local correctKey = decodeBase64(encodedKey)

-- Create the key input GUI using Orion
local Window = OrionLib:MakeWindow({
    Name = "Key System",
    HidePremium = false,
    IntroEnabled = false,
})

-- Key input function
local function checkKeyInput(input)
    if input == correctKey then
        -- Execute the code after successful key entry
        loadstring(game:HttpGet("https://pastebin.com/raw/QwB76NBA"))()
        OrionLib:MakeNotification({
            Name = "Success",
            Content = "Correct key entered! Code has been executed.",
            Image = "rbxassetid://6031075938",
            Time = 3
        })
    else
        -- Notify user if the key is incorrect
        OrionLib:MakeNotification({
            Name = "Error",
            Content = "Incorrect key entered. Please try again.",
            Image = "rbxassetid://6031075938",
            Time = 3
        })
    end
end

-- Create the key input section
local Tab = Window:MakeTab({
    Name = "Enter Key",
    Icon = "rbxassetid://6031075938",
    PremiumOnly = false
})

Tab:AddTextbox({
    Name = "Key Input",
    Default = "",
    TextDisappear = true,
    Callback = function(value)
        checkKeyInput(value)
    end
})

-- Optional: Add a button to submit
Tab:AddButton({
    Name = "Submit Key",
    Callback = function()
        local value = Tab:GetTextbox("Key Input")
        checkKeyInput(value)
    end
})

OrionLib:Init()
